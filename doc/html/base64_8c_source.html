58 /* Base64 encode IN array of size INLEN into OUT array of size OUTLEN.
59  If OUTLEN is less than BASE64_LENGTH(INLEN), write as many bytes as
60  possible. If OUTLEN is larger than BASE64_LENGTH(INLEN), also zero
61  terminate the output buffer. */
62 void base64_encode (const char *in, size_t inlen, char *out, size_t outlen) {
63  static const char b64str[64] =
64  "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
65 
66  while (inlen && outlen)
67  {
68  *out++ = b64str[(to_uchar (in[0]) >> 2) & 0x3f];
69  if (!--outlen)
70  break;
71  *out++ = b64str[((to_uchar (in[0]) << 4)
72  + (--inlen ? to_uchar (in[1]) >> 4 : 0))
73  & 0x3f];
74  if (!--outlen)
75  break;
76  *out++ =
77  (inlen
78  ? b64str[((to_uchar (in[1]) << 2)
79  + (--inlen ? to_uchar (in[2]) >> 6 : 0))
80  & 0x3f]
81  : '=');
82  if (!--outlen)
83  break;
84  *out++ = inlen ? b64str[to_uchar (in[2]) & 0x3f] : '=';
85  if (!--outlen)
86  break;
87  if (inlen)
88  inlen--;
89  if (inlen)
90  in += 3;
91  }
92 
93  if (outlen)
94  *out = '\0';
95 }

97 /* Allocate a buffer and store zero terminated base64 encoded data
98  from array IN of size INLEN, returning BASE64_LENGTH(INLEN), i.e.,
99  the length of the encoded data, excluding the terminating zero. On
100  return, the OUT variable will hold a pointer to newly allocated
101  memory that must be deallocated by the caller. If output string
102  length would overflow, 0 is returned and OUT is set to NULL. If
103  memory allocation failed, OUT is set to NULL, and the return value
104  indicates length of the requested memory block, i.e.,
105  BASE64_LENGTH(inlen) + 1. */
106 size_t base64_encode_alloc (const char *in, size_t inlen, char **out) {
107  size_t outlen = 1 + BASE64_LENGTH (inlen);
108 
109  /* Check for overflow in outlen computation.
110  *
111  * If there is no overflow, outlen >= inlen.
112  *
113  * If the operation (inlen + 2) overflows then it yields at most +1, so
114  * outlen is 0.
115  *
116  * If the multiplication overflows, we lose at least half of the
117  * correct value, so the result is < ((inlen + 2) / 3) * 2, which is
118  * less than (inlen + 2) * 0.66667, which is less than inlen as soon as
119  * (inlen > 4).
120  */
121  if (inlen > outlen)
122  {
123  *out = NULL;
124  return 0;
125  }
126 
127  *out = malloc (outlen);
128  if (!*out)
129  return outlen;
130 
131  base64_encode (in, inlen, *out, outlen);
132 
133  return outlen - 1;
134 }

292 /* Decode base64 encoded input array IN of length INLEN to output
293  array OUT that can hold *OUTLEN bytes. Return true if decoding was
294  successful, i.e. if the input was valid base64 data, false
295  otherwise. If *OUTLEN is too small, as many bytes as possible will
296  be written to OUT. On return, *OUTLEN holds the length of decoded
297  bytes in OUT. Note that as soon as any non-alphabet characters are
298  encountered, decoding is stopped and false is returned. This means
299  that, when applicable, you must remove any line terminators that is
300  part of the data stream before calling this function. */
301 bool base64_decode (const char *in, size_t inlen, char *out, size_t *outlen) {
302  size_t outleft = *outlen;
303 
304  while (inlen >= 2)
305  {
306  if (!isbase64 (in[0]) || !isbase64 (in[1]))
307  break;
308 
309  if (outleft)
310  {
311  *out++ = ((b64[to_uchar (in[0])] << 2)
312  | (b64[to_uchar (in[1])] >> 4));
313  outleft--;
314  }
315 
316  if (inlen == 2)
317  break;
318 
319  if (in[2] == '=')
320  {
321  if (inlen != 4)
322  break;
323 
324  if (in[3] != '=')
325  break;
326 
327  }
328  else
329  {
330  if (!isbase64 (in[2]))
331  break;
332 
333  if (outleft)
334  {
335  *out++ = (((b64[to_uchar (in[1])] << 4) & 0xf0)
336  | (b64[to_uchar (in[2])] >> 2));
337  outleft--;
338  }
339 
340  if (inlen == 3)
341  break;
342 
343  if (in[3] == '=')
344  {
345  if (inlen != 4)
346  break;
347  }
348  else
349  {
350  if (!isbase64 (in[3]))
351  break;
352 
353  if (outleft)
354  {
355  *out++ = (((b64[to_uchar (in[2])] << 6) & 0xc0)
356  | b64[to_uchar (in[3])]);
357  outleft--;
358  }
359  }
360  }
361 
362  in += 4;
363  inlen -= 4;
364  }
365 
366  *outlen -= outleft;
367 
368  if (inlen != 0)
369  return false;
370 
371  return true;
372 }

374 /* Allocate an output buffer in *OUT, and decode the base64 encoded
375  data stored in IN of size INLEN to the *OUT buffer. On return, the
376  size of the decoded data is stored in *OUTLEN. OUTLEN may be NULL,
377  if the caller is not interested in the decoded length. *OUT may be
378  NULL to indicate an out of memory error, in which case *OUTLEN
379  contains the size of the memory block needed. The function returns
380  true on successful decoding and memory allocation errors. (Use the
381  *OUT and *OUTLEN parameters to differentiate between successful
382  decoding and memory error.) The function returns false if the
383  input was invalid, in which case *OUT is NULL and *OUTLEN is
384  undefined. */
385  bool
386 base64_decode_alloc (const char *in, size_t inlen, char **out,
387  size_t *outlen)
388 {
389  /* This may allocate a few bytes too much, depending on input,
390  but it's not worth the extra CPU time to compute the exact amount.
391  The exact amount is 3 * inlen / 4, minus 1 if the input ends
392  with "=" and minus another 1 if the input ends with "==".
393  Dividing before multiplying avoids the possibility of overflow. */
394  size_t needlen = 3 * (inlen / 4) + 2;
395 
396  *out = malloc (needlen);
397  if (!*out)
398  return true;
399 
400  if (!base64_decode (in, inlen, *out, &needlen))
401  {
402  free (*out);
403  *out = NULL;
404  return false;
405  }
406 
407  if (outlen)
408  *outlen = needlen;
409 
410  return true;
411 }
