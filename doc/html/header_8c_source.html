54 static char *trim(char *str) {
55  size_t len = 0;
56  char *frontp = str;
57  char *endp = NULL;
58 
59  if ( unlikely(NULL == str) ) {
60  return NULL;
61  }
62 
63  if ( unlikely(str[0] == '\0') ) {
64  return str;
65  }
66 
67  len = strlen(str);
68  endp = str + len;
69 
70  /* Move the front and back pointers to address the first non-whitespace
71  * characters from each end.
72  */
73  while (isspace((unsigned char) *frontp)) {
74  ++frontp;
75  }
76 
77  if (endp != frontp) {
78  while (isspace((unsigned char) *(--endp)) && endp != frontp) {}
79  }
80 
81  if (str + len - 1 != endp) {
82  *(endp + 1) = '\0';
83  } else if (frontp != str && endp == frontp) {
84  *str = '\0';
85  }
86 
87  /* Shift the string so that it starts at str so that if it's dynamically
88  * allocated, we can still free it on the returned pointer. Note the reuse
89  * of endp to mean the front of the string buffer now.
90  */
91  endp = str;
92  if (frontp != str) {
93  while (*frontp) {
94  *endp++ = *frontp++;
95  }
96  *endp = '\0';
97  }
98 
99  return str;
100 }
