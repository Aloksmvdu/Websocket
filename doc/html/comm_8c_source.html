975  }

988 static void WSS_handshake(server_t *server, session_t *session, int id) {

990  header_t *header;
991  char buffer[server->config->size_buffer];
992  enum HttpStatus_Code code;
993  message_t *message;
994  bool ssl = false;
995 
996  memset(buffer, '\0', server->config->size_buffer);
997 
998 #ifdef USE_OPENSSL
999  if (NULL != server->ssl_ctx) {
1000  ssl = true;
1001  }
1002 #endif
1003 
1004  WSS_log_trace("Preparing client header");
1005 
1006  if ( unlikely(NULL != session->header) ) {
1007  header = session->header;
1008  session->header = NULL;
1009  } else {
1010  if ( unlikely(NULL == (header = WSS_malloc(sizeof(header_t)))) ) {
1011  return;
1012  }
1013 
1014  header->content = NULL;
1015  header->method = NULL;
1016  header->version = NULL;
1017  header->path = NULL;
1018  header->host = NULL;
1019  header->payload = NULL;
1020  header->length = 0;
1021  header->ws_version = 0;
1022  header->ws_type = UNKNOWN;
1023  header->ws_protocol = NULL;
1024  header->ws_upgrade = NULL;
1025  header->ws_connection = NULL;
1026  header->ws_extensions = NULL;
1027  header->ws_extensions_count = 0;
1028  header->ws_origin = NULL;
1029  header->ws_key = NULL;
1030  header->ws_key1 = NULL;
1031  header->ws_key2 = NULL;
1032  header->ws_key3 = NULL;
1033  }
1034 
1035  WSS_log_trace("Reading headers");
1036 
1037  // Continue reading until we get no bytes back
1038  do {
1039  n = WSS_read_internal(server, session, buffer);
1040 
1041  switch (n) {
1042  // Wait for IO for either read or write on the filedescriptor
1043  case -2:
1044  session->header = header;
1045  WSS_write_notify(server, session->fd);
1046  return;
1047  // An error occured, notify client by writing back to it
1048  case -1:
1049  WSS_log_trace("Rejecting HTTP request due to being unable to read from client");
1050 
1051  if ( likely(NULL != (message = WSS_http_response(header,
1052  HttpStatus_InternalServerError, "Unable to read from client"))) ) {
1053  WSS_write_internal(session, message, id);
1054  WSS_write_notify(server, session->fd);
1055  }
1056 
1057  WSS_free_header(header);
1058 
1059  return;
1060  case 0:
1061  break;
1062  default:
1063  // Reallocate space for the header and copy buffer into it
1064  if ( unlikely(NULL == (header->content = WSS_realloc((void **) &header->content, header->length*sizeof(char), (header->length+n+1)*sizeof(char)))) ) {
1065  WSS_log_fatal("Unable to realloc header content");
1066  return;
1067  }
1068  memcpy(header->content+header->length, buffer, n);
1069  header->length += n;
1070  memset(buffer, '\0', server->config->size_buffer);
1071 
1072  // Check if payload from client is too large for the server to handle.
1073  // If so write error back to the client
1074  if ( unlikely(header->length > (server->config->size_header+server->config->size_uri+server->config->size_payload)) ) {
1075  WSS_log_trace("Rejecting HTTP request as client payload is too large for the server to handle");
1076 
1077  if ( likely(NULL != (message = WSS_http_response(header, HttpStatus_PayloadTooLarge,
1078  "The given payload is too large for the server to handle"))) ) {
1079  WSS_write_internal(session, message, id);
1080  WSS_write_notify(server, session->fd);
1081  }
1082 
1083  WSS_free_header(header);
1084 
1085  return;
1086  }
1087  break;
1088  }
1089  } while ( likely(n != 0) );
1090 
1091  session->state = STALE;
1092 
1093  WSS_log_debug("Client header: \n%s", header->content);
1094 
1095  WSS_log_trace("Starting parsing header received from client");
1096 
1097  // Parsing HTTP header
1098  // If header could not be parsed correctly, notify client with response.
1099  if ( (code = WSS_parse_header(session->fd, header, server->config)) != HttpStatus_OK ) {
1100  WSS_log_trace("Rejecting HTTP request due to header not being correct");
1101 
1102  if ( likely(NULL != (message = WSS_http_response(header, code, "Unable to parse header"))) ) {
1103  WSS_write_internal(session, message, id);
1104  WSS_write_notify(server, session->fd);
1105  }
1106 
1107  WSS_free_header(header);
1108 
1109  return;
1110  }
1111 
1112  // Serve favicon
1113  if (strlen(header->path) == 12 && strncmp(header->path, "/favicon.ico", 12) == 0) {
1114  if (NULL != (message = WSS_favicon_response(header, code, server->config))) {
1115  WSS_log_trace("Serving a favicon to the client");
1116 
1117  // Find and serve favicon.
1118  WSS_write_internal(session, message, id);
1119  WSS_write_notify(server, session->fd);
1120  } else {
1121  WSS_log_trace("Rejecting HTTP request due to favicon not being available");
1122 
1123  // Else notify client that favicon could not be found
1124  code = HttpStatus_NotFound;
1125  if ( likely(NULL != (message = WSS_http_response(header, code, "File not found"))) ) {
1126  WSS_write_internal(session, message, id);
1127  WSS_write_notify(server, session->fd);
1128  }
1129  }
1130 
1131  WSS_free_header(header);
1132 
1133  return;
1134  }
1135 
1136  WSS_log_trace("Header successfully parsed");
1137 
1138  // Create Upgrade HTTP header based on clients header
1139  code = WSS_upgrade_header(header, server->config, ssl, server->port);
1140  switch (code) {
1141  case HttpStatus_UpgradeRequired:
1142  WSS_log_trace("Rejecting HTTP request as the service requires use of the Websocket protocol.");
1143  message = WSS_upgrade_response(header, code,
1144  "This service requires use of the Websocket protocol.");
1145  break;
1146  case HttpStatus_NotImplemented:
1147  WSS_log_trace("Rejecting HTTP request as Websocket protocol is not yet implemented");
1148  message = WSS_http_response(header, code,
1149  "Websocket protocol is not yet implemented");
1150  break;
1151  case HttpStatus_SwitchingProtocols:
1152  message = WSS_handshake_response(header, code);
1153  break;
1154  case HttpStatus_NotFound:
1155  WSS_log_trace("Rejecting HTTP request as the page requested was not found.");
1156  message = WSS_http_response(header, code,
1157  "The page requested was not found.");
1158  break;
1159  case HttpStatus_Forbidden:
1160  WSS_log_trace("Rejecting HTTP request as the origin is not allowed to establish a websocket connection.");
1161  message = WSS_http_response(header, code,
1162  "The origin is not allowed to establish a websocket connection.");
1163  break;
1164  default:
1165  WSS_log_trace("Rejecting HTTP request as server was unable to parse http header as websocket request");
1166  message = WSS_http_response(header, code,
1167  "Unable to parse http header as websocket request");
1168  break;
1169  }
1170 
1171  // If code status isnt switching protocols, we notify client with a HTTP error
1172  if (code != HttpStatus_SwitchingProtocols) {
1173  WSS_log_trace("Unable to establish websocket connection");
1174 
1175  WSS_write_internal(session, message, id);
1176  WSS_write_notify(server, session->fd);
1177 
1178  WSS_free_header(header);
1179 
1180  return;
1181  }
1182 
1183  // Use echo protocol if none was chosen
1184  if (NULL == header->ws_protocol) {
1185  header->ws_protocol = find_subprotocol("echo");
1186  }
1187 
1188  // Notify websocket protocol of the connection
1189  header->ws_protocol->connect(session->fd);
1190 
1191  // Set session as fully handshaked
1192  session->handshaked = true;
1193  session->header = header;
1194 
1195  WSS_write_internal(session, message, id);
1196  WSS_write_notify(server, session->fd);
1197 }
