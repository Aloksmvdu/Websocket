21  *
22  * Description:
23  * This file implements the Secure Hashing Standard as defined
24  * in FIPS PUB 180-1 published April 17, 1995.
25  *
26  * The Secure Hashing Standard, which uses the Secure Hashing
27  * Algorithm (SHA), produces a 160-bit message digest for a
28  * given data stream. In theory, it is highly improbable that
29  * two messages will produce the same message digest. Therefore,
30  * this algorithm can serve as a means of providing a "fingerprint"
31  * for a message.
32  *
33  * Portability Issues:
34  * SHA-1 is defined in terms of 32-bit "words". This code was
35  * written with the expectation that the processor has at least
36  * a 32-bit machine word size. If the machine word size is larger,
37  * the code should still function properly. One caveat to that
38  * is that the input functions taking characters and character
39  * arrays assume that only 8 bits of information are stored in each
40  * character.
41  *
42  * Caveats:
43  * SHA-1 is designed to work with messages less than 2^64 bits
44  * long. Although SHA-1 allows a message digest to be generated for
45  * messages of any number of bits less than 2^64, this
46  * implementation only works with messages with a length that is a
47  * multiple of the size of an 8-bit character.
48  *
49  */

54  * Define the circular shift macro
55  */
56 #define SHA1CircularShift(bits,word) \
57  ((((word) << (bits)) & 0xFFFFFFFF) | \
58  ((word) >> (32-(bits))))

98  * SHA1Result
99  *
100  * Description:
101  * This function will return the 160-bit message digest into the
102  * Message_Digest array within the SHA1Context provided
103  *
104  * Parameters:
105  * context: [in/out]
106  * The context to use to calculate the SHA-1 hash.

109  * 1 if successful, 0 if it failed.
110  *
111  * Comments:
112  *
113  */
114 int SHA1Result(SHA1Context *context)
115 {
116 
117  if (context->Corrupted)
118  {
119  return 0;
120  }
121 
122  if (!context->Computed)
123  {
124  SHA1PadMessage(context);
125  context->Computed = 1;
126  }
127 
128  return 1;

131 /*
132  * SHA1Input
133  *
134  * Description:
135  * This function accepts an array of octets as the next portion of
136  * the message.
137  *
138  * Parameters:
139  * context: [in/out]
140  * The SHA-1 context to update
141  * message_array: [in]
142  * An array of characters representing the next portion of the
143  * message.
144  * length: [in]
145  * The length of the message in message_array
146  *
147  * Returns:
148  * Nothing.
149  *
150  * Comments:
151  *
152  */
153 void SHA1Input( SHA1Context *context,
154  const unsigned char *message_array,
155  unsigned length)
156 {
157  if (!length)
158  {
159  return;
160  }
161 
162  if (context->Computed || context->Corrupted)
163  {
164  context->Corrupted = 1;
165  return;
166  }
167 
168  while(length-- && !context->Corrupted)
169  {
170  context->Message_Block[context->Message_Block_Index++] =
171  (*message_array & 0xFF);
172 
173  context->Length_Low += 8;
174  /* Force it to 32 bits */
175  context->Length_Low &= 0xFFFFFFFF;
176  if (context->Length_Low == 0)
177  {
178  context->Length_High++;
179  /* Force it to 32 bits */
180  context->Length_High &= 0xFFFFFFFF;
181  if (context->Length_High == 0)
182  {
183  /* Message is too long */
184  context->Corrupted = 1;
185  }
186  }
187 
188  if (context->Message_Block_Index == 64)
189  {
190  SHA1ProcessMessageBlock(context);
191  }
192 
193  message_array++;
194  }
195 }

197 /*
198  * SHA1ProcessMessageBlock
199  *
200  * Description:
201  * This function will process the next 512 bits of the message
202  * stored in the Message_Block array.
203  *
204  * Parameters:
205  * None.
206  *
207  * Returns:
208  * Nothing.
209  *
210  * Comments:
211  * Many of the variable names in the SHAContext, especially the
212  * single character names, were used because those were the names
213  * used in the publication.
214  *
215  *
216  */
217 void SHA1ProcessMessageBlock(SHA1Context *context)
218 {
219  const unsigned K[] = /* Constants defined in SHA-1 */
220  {
221  0x5A827999,
222  0x6ED9EBA1,
223  0x8F1BBCDC,
224  0xCA62C1D6
225  };
226  int t; /* Loop counter */
227  unsigned temp; /* Temporary word value */
228  unsigned W[80]; /* Word sequence */
229  unsigned A, B, C, D, E; /* Word buffers */
230 
231  /*
232  * Initialize the first 16 words in the array W
233  */
234  for(t = 0; t < 16; t++)
235  {
236  W[t] = ((unsigned) context->Message_Block[t * 4]) << 24;
237  W[t] |= ((unsigned) context->Message_Block[t * 4 + 1]) << 16;
238  W[t] |= ((unsigned) context->Message_Block[t * 4 + 2]) << 8;
239  W[t] |= ((unsigned) context->Message_Block[t * 4 + 3]);
240  }
241 
242  for(t = 16; t < 80; t++)
243  {
244  W[t] = SHA1CircularShift(1,W[t-3] ^ W[t-8] ^ W[t-14] ^ W[t-16]);
245  }
246 
247  A = context->Message_Digest[0];
248  B = context->Message_Digest[1];
249  C = context->Message_Digest[2];
250  D = context->Message_Digest[3];
251  E = context->Message_Digest[4];
252 
253  for(t = 0; t < 20; t++)
254  {
255  temp = SHA1CircularShift(5,A) +
256  ((B & C) | ((~B) & D)) + E + W[t] + K[0];
257  temp &= 0xFFFFFFFF;
258  E = D;
259  D = C;
260  C = SHA1CircularShift(30,B);
261  B = A;
262  A = temp;
263  }
264 
265  for(t = 20; t < 40; t++)
266  {
267  temp = SHA1CircularShift(5,A) + (B ^ C ^ D) + E + W[t] + K[1];
268  temp &= 0xFFFFFFFF;
269  E = D;
270  D = C;
271  C = SHA1CircularShift(30,B);
272  B = A;
273  A = temp;
274  }
275 
276  for(t = 40; t < 60; t++)
277  {
278  temp = SHA1CircularShift(5,A) +
279  ((B & C) | (B & D) | (C & D)) + E + W[t] + K[2];
280  temp &= 0xFFFFFFFF;
281  E = D;
282  D = C;
283  C = SHA1CircularShift(30,B);
284  B = A;
285  A = temp;
286  }
287 
288  for(t = 60; t < 80; t++)
289  {
290  temp = SHA1CircularShift(5,A) + (B ^ C ^ D) + E + W[t] + K[3];
291  temp &= 0xFFFFFFFF;
292  E = D;
293  D = C;
294  C = SHA1CircularShift(30,B);
295  B = A;
296  A = temp;
297  }
298 
299  context->Message_Digest[0] =
300  (context->Message_Digest[0] + A) & 0xFFFFFFFF;
301  context->Message_Digest[1] =
302  (context->Message_Digest[1] + B) & 0xFFFFFFFF;
303  context->Message_Digest[2] =
304  (context->Message_Digest[2] + C) & 0xFFFFFFFF;
305  context->Message_Digest[3] =
306  (context->Message_Digest[3] + D) & 0xFFFFFFFF;
307  context->Message_Digest[4] =
308  (context->Message_Digest[4] + E) & 0xFFFFFFFF;
309 
310  context->Message_Block_Index = 0;
311 }

313 /*
314  * SHA1PadMessage
315  *
316  * Description:
317  * According to the standard, the message must be padded to an even
318  * 512 bits. The first padding bit must be a '1'. The last 64
319  * bits represent the length of the original message. All bits in
320  * between should be 0. This function will pad the message
321  * according to those rules by filling the Message_Block array
322  * accordingly. It will also call SHA1ProcessMessageBlock()
323  * appropriately. When it returns, it can be assumed that the
324  * message digest has been computed.
325  *
326  * Parameters:
327  * context: [in/out]
328  * The context to pad
329  *
330  * Returns:
331  * Nothing.
332  *
333  * Comments:
334  *
335  */
336 void SHA1PadMessage(SHA1Context *context)
337 {
338  /*
339  * Check to see if the current message block is too small to hold
340  * the initial padding bits and length. If so, we will pad the
341  * block, process it, and then continue padding into a second
342  * block.
343  */
344  if (context->Message_Block_Index > 55)
345  {
346  context->Message_Block[context->Message_Block_Index++] = 0x80;
347  while(context->Message_Block_Index < 64)
348  {
349  context->Message_Block[context->Message_Block_Index++] = 0;
350  }
351 
352  SHA1ProcessMessageBlock(context);
353 
354  while(context->Message_Block_Index < 56)
355  {
356  context->Message_Block[context->Message_Block_Index++] = 0;
357  }
358  }
359  else
360  {
361  context->Message_Block[context->Message_Block_Index++] = 0x80;
362  while(context->Message_Block_Index < 56)
363  {
364  context->Message_Block[context->Message_Block_Index++] = 0;
365  }
366  }
367 
368  /*
369  * Store the message length as the last 8 octets
370  */
371  context->Message_Block[56] = (context->Length_High >> 24) & 0xFF;
372  context->Message_Block[57] = (context->Length_High >> 16) & 0xFF;
373  context->Message_Block[58] = (context->Length_High >> 8) & 0xFF;
374  context->Message_Block[59] = (context->Length_High) & 0xFF;
375  context->Message_Block[60] = (context->Length_Low >> 24) & 0xFF;
376  context->Message_Block[61] = (context->Length_Low >> 16) & 0xFF;
377  context->Message_Block[62] = (context->Length_Low >> 8) & 0xFF;
378  context->Message_Block[63] = (context->Length_Low) & 0xFF;
379 
380  SHA1ProcessMessageBlock(context);
381 }
